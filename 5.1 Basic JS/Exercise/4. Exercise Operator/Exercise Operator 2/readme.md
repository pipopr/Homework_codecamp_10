        2.1. จงหาผลลัพธ์ของ Statement ต่อไปนี้
            "" + 1 + 0
            "" - 1 + 0
            true + false
            6 / "3"
            "2" * "3"
            4 + 5 + "px"
            "$" + 4 + 5
            "4" - 2
            "4px" - 2
            7 / 0
            "  -9  " + 5
            "  -9  " - 5
            null + 1
            undefined + 1
            " \t \n" - 2

            answer
            "" + 1 + 0      ===> 10
            "" - 1 + 0      ===> -1
            true + false    ===> 1
            6 / "3"         ===> 2
            "2" * "3"       ===> 6
            4 + 5 + "px"    ===> 9px
            "$" + 4 + 5     ===> $45
            "4" - 2         ===> 2
            "4px" - 2       ===> Nan
            7 / 0           ===> Infinity
            "  -9  " + 5    ===>   -9  5
            "  -9  " - 5    ===> -14
            null + 1        ===> 1
            undefined + 1   ===> Nan
            " \t \n" - 2    ===> -2
